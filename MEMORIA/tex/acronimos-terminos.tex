\newglossaryentry{middleware}{
	name={\it Middleware},
	description={Software que se encuentra entre el sistema operativo y las aplicaciones que funcionan sobre él}
}
\newglossaryentry{recomendaciones}{
  name={\it Sistema de recomendaciones},
  description={Mecanismo que sugiere elementos (proyectos o usuarios) en función de similitudes, coincidencias o preferencias}
}
\newglossaryentry{fullstack}{
  name={\it Full Stack},
  description={Perfil de desarrollador que trabaja tanto en el desarrollo del lado cliente (frontend) como en el servidor (backend) de una aplicación web}
}
\newglossaryentry{frontend}{
  name={\it Frontend},
  description={Parte de una aplicación web que interactúa directamente con el usuario}
}
\newglossaryentry{backend}{
  name={\it Backend},
  description={Parte de una aplicación web que se encarga de la lógica del servidor y la gestión de datos}
}
\newglossaryentry{match}{
    name={\it Match},
    plural=Matches,
    description={Coincidencia entre dos perfiles en un sistema de emparejamiento digital, determinada por afinidad entre características, 
                 intereses o competencias. Puede generarse de forma automática o mediante la validación mutua de las partes implicadas, 
                 y permite habilitar la interacción entre ellas}
}
